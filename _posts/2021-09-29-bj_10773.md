```java
import java.util.*;

public class Main {
    public int solution(int n, int[] arr){
        Stack<Integer> stack = new Stack<>();
        int sum = 0;

        for(int i = 0; i < n; i++){
            if(arr[i]!= 0){
                stack.push(arr[i]);
            }else {
                stack.pop();
            }
        }

        while(!stack.isEmpty()){
            sum += stack.pop();
        }
        return sum;
    }

    public static void main(String[] args) {
        Main function = new Main();
        Scanner sc = new Scanner(System.in);
        int n = sc.nextInt();
        int[] arr = new int[n];
        for(int i = 0; i < n; i++){
            arr[i] = sc.nextInt();
        }
        System.out.println(function.solution(n, arr));
    }
}
```
