```java
import java.util.*;

public class Main {
    public static void main(String[] args){
        Scanner sc = new Scanner(System.in);
        int input = sc.nextInt();
        int[] array = new int[input];

        for(int i = 0; i < input; i++){
            array[i] = sc.nextInt();
        }

        sc.close();

        int min = array[0];
        int max = array[0];

        for(int i = 0; i< input; i++){
            if(min > array[i]){
                min = array[i];
            }
        }
        for(int i = 0; i< input; i++){
            if(max < array[i]){
                max = array[i];
            }
        }
        System.out.println(min +" "+ max);
    }
}
```
